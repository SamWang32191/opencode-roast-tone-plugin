> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the single-file `roast.ts` script into a publishable npm package that OpenCode can install via `opencode plugin opencode-roast-tone-plugin@latest`.

**Architecture:** Keep the plugin server-only and split it into a tiny `src/server.ts` entry plus `src/tone.ts` for the prompt constant. Package the compiled JS from `dist/`, expose it through both `exports["./server"]` and `main`, document CLI-first installation, and keep local path-based development support.

**Tech Stack:** npm, TypeScript (`tsc`), Vitest, OpenCode plugin API

type MessagePart = { type: string; text: string };
type Message = { info: { role: string }; parts: MessagePart[] };
type TransformOutput = { messages: Message[] };

export default function createRoastPlugin() {
  return {
    "experimental.chat.messages.transform": async (_input: unknown, output: TransformOutput) => {
      const firstUser = output.messages.find((message) => message.info.role === "user");
      if (!firstUser) return;

      const ref = firstUser.parts[0];
      firstUser.parts.unshift({ ...ref, type: "text", text: TONE });

  };
}

Update `src/server.ts`:

```ts
import { TONE } from "./tone";

type MessagePart = { type: string; text: string };
type Message = { info: { role: string }; parts: MessagePart[] };
type TransformOutput = { messages: Message[] };

export default function createRoastPlugin() {
  return {
    "experimental.chat.messages.transform": async (_input: unknown, output: TransformOutput) => {
      if (!output.messages.length) return;

      const firstUser = output.messages.find((message) => message.info.role === "user");
      if (!firstUser || !firstUser.parts.length) return;

      if (firstUser.parts.some((part) => part.type === "text" && part.text.includes(TONE))) return;

      const ref = firstUser.parts[0];
      firstUser.parts.unshift({ ...ref, type: "text", text: TONE });
    }
  };
}
```
